# Chú thích các placeholder:
# {order_number} - Số đơn hàng
# {order_date} - Ngày đặt hàng
# {customer_name} - Tên khách hàng
# {tax_code} - Mã số thuế
# {invoice_address} - Địa chỉ xuất hóa đơn
# {recipient_name} - Tên người nhận
# {recipient_phone} - Số điện thoại người nhận
# {delivery_address} - Địa chỉ giao hàng
# {#products} - Bắt đầu vòng lặp sản phẩm
# {/products} - Kết thúc vòng lặp sản phẩm
# {stt} - Số thứ tự
# {ten_hang_hoa} - Tên hàng hóa
# {mo_ta} - Mô tả sản phẩm
# {dvt} - Đơn vị tính
# {so_luong} - Số lượng
# {don_gia} - Đơn giá
# {thanh_tien} - Thành tiền
# {subtotal} - Tổng tiền hàng
# {tax_amount} - Tiền thuế GTGT
# {grand_total} - Tổng cộng
# {total_in_words} - Số tiền bằng chữ
